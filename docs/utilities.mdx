{Object.entries(props.utilities).map(([key, utility]) => (
<div className="mb-16" key={key}>
    <div className="text-2xl font-bold mb-4">
        <a href={`#${key}`} name={key} className="text-gray-800 hover:text-gray-900">{key}</a>
    </div>
    <div className="mb-4">
        {utility.attributes.description}
    </div>
    {(utility.variants?.length > 0) && (
        <div className="mb-4 flex gap-2 flex-wrap">
        {utility.variants.map(variant => (
            <Fragment key={variant}>
            {variant === "default" && (
                <div className="py-1 px-2 bg-gray-800 text-white text-xs rounded-md">
                    <span className="font-bold">default</span>
                </div>
            )}
            {variant === "responsive" && (
                <div className="py-1 px-2 bg-gray-600 text-white text-xs rounded-md">
                    <span className="font-bold">responsive</span>
                </div>
            )}
            {variant !== "responsive" && variant !== "default" && (
                <div className="py-1 px-2 bg-gray-300 text-xs rounded-md">
                    <span>pseudo: <span className="font-bold">{variant}</span></span>
                </div>
            )}
            </Fragment>
        ))}
        </div>
    )}
    <div className="max-h-96 overflow-auto text-gray-300">
        <table className="w-full">
            {Object.entries(utility.values).map(item => (
            <tr key={item[0]} className="border border-gray-300 text-xs">
                <td className="p-4 font-mono bg-gray-200 text-blue-600">
                    {utility.classname && item[0] !== "DEFAULT" && (
                        <span>{utility.classname}-{item[0]}</span>
                    )}
                    {!utility.classname && item[0] !== "DEFAULT" && (
                        <span>{item[0]}</span>
                    )}
                    {utility.classname && item[0] === "DEFAULT" && (
                        <span>{utility.classname}</span>
                    )}
                </td>
                <td className="p-4 font-mono text-gray-700">
                {utility.properties.map(p => (
                    <div key={p} className="mb-1">
                        <span className="text-indigo-700">{p}</span>: {item[1]};
                    </div>
                ))}    
                </td>
            </tr>
            ))}
        </table>
    </div>
</div>
))}
